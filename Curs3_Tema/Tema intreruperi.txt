EX1

#include <avr/io.h>
#include <avr/interrupt.h>
#define F_CPU 16000000UL
#include <util/delay.h>

ISR(INT0_vect)
{
	if (PIND & (1 << PD0)) {
		PORTA &= ~(1 << PA1);
		} else {
		PORTA |= (1 << PA1);
	}
}

int main(void)
{

	DDRA |= (1 << PA0) | (1 << PA1);
	DDRD &= ~(1 << PD0);
	PORTD |= (1 << PD0);
	
	cli();
	EICRA |= (1 << ISC00);
	EICRA &= ~(1 << ISC01);
	EIMSK |= (1 << INT0);
	sei();
	
	while (1)
	{
		PORTA ^= (1 << PA0);
		_delay_ms(3000);
	}
}

#include <avr/io.h>
#include <avr/interrupt.h>
#define F_CPU 16000000UL
#include <util/delay.h>

ISR(INT0_vect)
{
	PORTA |= (1 << PA0);
}

ISR(INT1_vect)
{
	PORTA &= ~((1 << PA0));
}

int main(void)
{
	DDRA |= (1 << PA0);
	PORTA |= (1 << PA0);
	
	DDRD &= ~((1 << PD0) | (1 << PD1));
	PORTD |= (1 << PD0) | (1 << PD1);
	
	cli();
	EICRA |= (1 << ISC01) | (1 << ISC11);
	EICRA &= ~((1 << ISC00) | (1 << ISC10));
	EIMSK |= (1 << INT0) | (1 << INT1);
	sei();
	
	while (1)
	{
	}
}
